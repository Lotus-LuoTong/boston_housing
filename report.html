def performance_metric2(y_true, y_predict):
    """"""
    true_mean = np.mean(y_true)
    tot = [(x - true_mean) ** 2 for x in y_true]
    res = [(x - y) ** 2 for (x,y) in zip(y_predict, y_true)] 
    ss_tot = sum(tot)
    ss_res = sum(res)
    score = 1 - float(ss_res) / ss_tot

    return score

def cross_val_score(estimator, X, y, scoring = performance_metric, cv=3):
    """  """
    import random
    #
    for i in range(cv):
        locals()['X' + str(i)] = []
        locals()['y' + str(i)] = []
    #
    for j in range(len(X)):
        i = random.sample(range(cv),1)[0]
        locals()['X' + str(i)].append(X[i])
        locals()['y' + str(i)].append(y[i])

    for i in range(cv):
        X_train = []
        y_train = []
        X_test = locals()['X' + str(i)]
        y_test = locals()['y' + str(i)]
        for j in range(cv):
            if(i != j):
                X_train += locals()['X' + str(j)]
                y_train += locals()['y' + str(j)]
        estimator.fit(X_train, y_train)
        pred = estimator.predict(X_test)
        scores[i] = scoring(y_test, pred)
    scores = [0,0,0]
    return scores
